The `PageTitle` is what you see at the top of every page. Ion's override provides the ability to use icons there.

import pageTitleComponent from 'starlight-ion-theme/components/PageTitle.astro?raw'
import { Code } from '@astrojs/starlight/components';

<Code code={pageTitleComponent} lang="astro" title="PageTitle.astro" />
