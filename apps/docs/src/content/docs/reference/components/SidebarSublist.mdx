Ion's `SidebarSublist` override allows you to use Icons in the sidebar.

import sidebarSublistComponent from 'starlight-ion-theme/components/SidebarSublist.astro?raw'
import { Code } from '@astrojs/starlight/components';

<Code code={sidebarSublistComponent} lang="astro" title="SidebarSublist.astro" />
